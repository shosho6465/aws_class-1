		/* 여러줄 주석 */
		/* 
		변수 선언 방법
		방법1.
		var 변수명 = 값;
		var 변수명;

		방법2.
		let 변수명 = 값;
		let 변수명;

		변수
		- 값을 저장하는 공간
		- 예 
			- 1000
			- 볼펜값 : 1000
		
		var와 let의 차이
		- var는 호이스팅이 일어나지만, let은 일어나지 않음
		- var는 동일한 변수를 중복선언해도 동작, let은 예외 발생
		- let 사용을 추천
		*/		
		/*
		호이스팅
		- 페이지 로딩 시 변수나 함수가 위로 올려지는 동작
		- var로 선언된 변수,함수는 호이스팅이 적용
		*/
		let age = 20;
		//let age = 20; //예외 발생
		var name = "홍길동";
		var name = "홍길동";

		//count = 10; //선언전 초기화. 예외 발생
		let count; //변수 선언

		count2 = 20; //선언전 초기화
		var count2; //변수 선언
		/*
		39~48줄이 로딩되면 다음과 같이 됨
		var name;
		var count2;

		let age = 20;//39줄
		//let age = 20;//39줄
		name = "홍길동"; //41줄
		name = "홍길동"; //42줄
		//count = 10; //44줄
		let count; //45줄
		count2 = 20;//47줄
		*/
		/* 
		상수 선언 : 값을 한번 지정하면 바꿀수 없는 것. 
		- 상수는 보통 대문자로
		- 여러 단어로 이루어진 경우 _를 이용
		선언 방법
		- const 상수명 = 값;
		*/
		const PI = 3.14;
		//PI = 3.14;//상수는 재할당이 안됨
		/*
		변수명 작성 관례
		- 변수명은 의미있게 작성. 
			- 잘못된예 : a(사과의 갯수)
			- 잘된예 : appleCount
		- 카멜표기법
			- 두단어 이상으로 된 변수명의 두번째 단어부터 첫글자를 대문자로

		변수명(상수) 작성 규칙
		- 변수명은 숫자로 시작할 수 없음
		- 변수명에 특수문자는 _와 $만 가능
		- 예약어(키워드) 사용 불가
			- 예약어 : 기능이 있는 단어. const, let, class등
		- 변수명은 대소문자를 구별
			- num와 NUM는 다른 변수
		*/
